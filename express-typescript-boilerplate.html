Structure du project

    
      Projet/ 
      │
      └───src/
      │   │   Server.ts
      │   │   index.ts
      │   │
      │   └───Controller/
      │   |   │   BaseController.ts // Controlleur abstrait (parent)
      │   |   │   HelloController.ts // Ex Controlleur
      |   |
      │   └───Core/
      │   |   │   Core.ts // Application Core
      │   |   │   DependencyInjection.ts // Injection de dependence et utilité d'importation
      |   |
      │   └───Middlewares/
      │   |   │   ErrorMiddleware.ts // Middleware expressjs retournant du json si exception il y a
      |   |
      │   └───Services/
      │   |   │   FakeService.ts // Ex service
      │   |   │   HelloService.ts // Ex service
      |   |
      │   └───Utils/
      │   |   │   ApiError.ts // Classe d'erreur pérsonalisée
      │   |   │   GlobalMessage.ts // Get error object based on an errorCode
      |   |   
      │   └───config/
      │       │   config.ts // Configuration globale
      │       │   configDev.ts // Configuration dev. NODE_ENV=dev
      │       │   configProd.ts // Configuration prod. NODE_ENV=prod
      │       │   configTest.ts // Configuration test. NODE_ENV=test
      │       │   middlewares.ts // Configuration des middlewares
      │       │   parameters.ts // Application parameters
      │       │   routing.ts // Routing de l'application
      │       │   services.ts // Configuration des services
      │   
      └───Resources/
      |   │   errors.json // Message d'erreur global
      |
      └───test // Unitests
      └───dist // Fichiers dist

Serveur

    
      import http from 'http';
      import { Server } from './Server';

      Server.boostrap()
          .then((server) => {
              // Créer un serveur
              const httpServer = http.createServer(app);

              // Ecouter sur le port 3000
              // Vous pouvez changer le port avec la variable env NODE_PORT=8080
              httpServer.listen(server.getPort());
          });

Si vous préferez async / await 

    
      import http from 'http';
      import { Server } from './Server';

      ...

      const server = await Server.bootstrap();

      // Create server
      const httpServer = http.createServer(app);

      // Start listening on port 3000
      // You can override port by using NODE_PORT=8080
      httpServer.listen(server.getPort());
